def _chia_doi(self): # 1. Rùa và Thỏ xuất phát từ vạch đích (head) con_tro_cham = self.head con_tro_nhanh = self.head # 2. Vòng lặp: chạy tới khi Thỏ tới cuối đường while con_tro_nhanh.next is not None and con_tro_nhanh.next.next is not None: con_tro_cham = con_tro_cham.next # Rùa đi 1 ô con_tro_nhanh = con_tro_nhanh.next.next # Thỏ nhảy 2 ô # 3. Cắt đứt danh sách (ngay sau lưng Rùa) nua_dau = self.head nua_sau = con_tro_cham.next con_tro_cham.next = None if nua_sau is not None: nua_sau.truoc = None return (nua_dau, nua_sau)

def _tron(self, ds_a, ds_b, tieu_chi): node_tam = Node(None) # Node giả để làm mỏ neo hien_tai = node_tam node_a = ds_a.head node_b = ds_b.head # Trong khi cả 2 hàng đều còn người... while node_a is not None and node_b is not None: gia_tri_a = self._lay_gia_tri(node_a.data, tieu_chi) gia_tri_b = self._lay_gia_tri(node_b.data, tieu_chi) if gia_tri_a <= gia_tri_b: hien_tai.next = node_a node_a.truoc = hien_tai node_a = node_a.next else: hien_tai.next = node_b node_b.truoc = hien_tai node_b = node_b.next hien_tai = hien_tai.next # Vét nốt hàng còn sót lại if node_a is not None: hien_tai.next = node_a node_a.truoc = hien_tai self.cuoi = ds_a.cuoi if node_b is not None: hien_tai.next = node_b node_b.truoc = hien_tai self.cuoi = ds_b.cuoi # Bỏ node giả và chốt lại head self.head = node_tam.next if self.head is not None: self.head.truoc = None
